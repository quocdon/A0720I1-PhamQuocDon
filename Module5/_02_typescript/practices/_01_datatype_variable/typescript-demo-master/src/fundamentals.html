let message: string;
let total: number = 100;
let isProduction = true;
let prices: Array<number> = [120, 88, 60];
let languages: string[] = ['vi', 'en-us'];
let now = new Date();
let unknown: any;

enum Direction {
    UP,
    DOWN,
    LEFT,
    RIGHT
};

function log(msg: string): void {
    console.log(msg)
}

interface IPost {
    id: string;
    title: string;
    body?: string;
}

isProduction = false;
unknown = Direction.UP;
unknown = 'changed';

const post: IPost = {

};
message = 50;

Dòng 1 khởi tạo biến message kiểu string

Dòng 3 khởi tạo biến isProduction, với việc TypeScript tự động tính toán kiểu dữ liệu cho biến đó là boolean vì vế bên phải của phép gán là 1 giá trị kiểu boolean, quá trình này gọi là type inference

Dòng 4 và 5 là các cách tương đương để tạo 1 mảng

Dòng 7 khởi tạo biến unknown với kiểu any tức là biến này có thể được gán với bất kỳ kiểu dữ liệu nào.
P/s: Nên hạn chế dùng any

Dòng 9 đến 14 khởi tạo 1 enum có thể coi như 1 object dạng key-value

Dòng 16 khai báo function tên là log, void type có nghĩa rằng hàm này không return về giá gì gì hay return về undefined

Dòng 20 đến 24 khởi tạo một interface - một kiểu dữ liệu không có trong JavaScript, nhưng nó tồn tại trong TypeScript. Được dùng để khởi tạo hình dạng cho một object, giúp quá trình development có thể dễ đoán được trước lỗi, mà không generate ra phần code tương ứng ở JS, một phần nữa sẽ giúp các IDE/Editor gợi ý code tốt hơn

Dòng 30 đến 32 khởi tạo object có kiểu tương ứng với những gì IPost đã định nghĩa, nhưng vì object được gán vào không phù hợp, nên sẽ sinh ra lỗi:
Type '{}' is not assignable to type 'IPost'.
Property 'id' is missing in type '{}'.

Dòng 33 sẽ sinh ra lỗi: Type '50' is not assignable to type 'string'.
